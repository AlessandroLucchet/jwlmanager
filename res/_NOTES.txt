in ui....py file, add:

import os, import sys

    def resource_path(self, relative_path):
        try:
            base_path = sys._MEIPASS
        except Exception:
            base_path = os.path.abspath(".")
        return os.path.join(base_path, relative_path)


replace all u"icons/(.*?)" with self.resource_path\('icons/$1'\)
